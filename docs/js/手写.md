## **手写new操作符**

`new`实现了什么?

+ 创建了全新的对象
+ `__proto__`的指向
+ this指向新创建的对象

```javascript
function New(className){
  let res = {}
  if(className.prototype !== null){
    res.__proto__ = className.prototype
  }
  
  let ret = className.apply(res, Array.from(arguments).slice(1))
  if((typeof ret === 'object' || typeof ret === 'function') && ret !== null ) {
    return ret
  }
  return res
}
```

## **手写call**

`call` 改变了this的指向,直接调用了函数

```javascript
Function.prototype.call2 = function(thisArg = window){
 	thisArg.fn = this; //this谁调用指向谁
  let args = [...arguments].slice(1)
  let result = thisArg.fn(...args)
  delete thisArg.fn;
  return result;
}
```

### **手写Apply**

```javascript
Function.prototype.apply2 = function(thisArg = window){
  thisArg.fn = this;
  let result
  if(arguments[1]){
    result = thisArg.fn(...arguments[1])
  } else {
    result = thisArg.fn()
  }
  delete thisArg.fn
  return result;
}
```

### **instanceOf**

```
function instanceOf(left, right){
  let proto = left.__proto__;
  let prototype = right.prototype;
  while(true){
    if(proto == null) return false
    if(proto == prototype) return true
    proto = proto.__proto__
  }
}
```

### **Promise**

```javascript
const PENDDING = "pendding"
const FULFILLED = "fulfilled"
cosnt REJECTED = "rejected"

function Promise(excutor){
  let that = this;
  that.status = PENDDING
  that.value = undefined
  that.reason = undefined
  that.onFulfilledCallbacks = []
  that.onRejectedCallbacks = []
  // 直接转resolve
  function resolve(value){
    if(value instanceof Promise){
      return value.then(resolve, reject)
    }
    setTimeout(() => {
      if(that.status == PENDDING){
        that.status = FULFILLED
        that.value= value
        that.onFulfilledCallbacks.forEach(cb => cb(that.value))
      }
    })
  }
  
  function reject(reason){
    setTimeout(() => {
            // 调用reject 回调对应onRejected函数
      if (that.status === PENDING) {
        // 只能由pending状态 => rejected状态 (避免调用多次resolve reject)
        that.status = REJECTED;
        that.reason = reason;
        that.onRejectedCallbacks.forEach(cb => cb(that.reason));
      }
    })
  }
  
  try{
    excutor(resolve, reject)
  } catch(err){
    reject(err)
  }
  
  Promise.prototype.then = function(){}
  Promise.prototype.catch = function(){}
}
```

### **EventBus**

随Vue声明周期的EventBus

```javascript
class EventBus{
  constructor(vue){
    if(!this.handles){
      Object.defineProperty(this, 'handles', {
        value: {},
        enumerable: false
      })
    }
    this.Vue = vue
    this.eventMapUid = {}
  }
  
  setEventMapUid(uid, eventName){
    if(!this.eventMapUid[uid]) this.eventMapUid[uid] = []
    this.eventMapUid[uid].push(eventName)
  }
  
  $on(eventName, callback,vm){
    if(!this.handles[eventName]) this.handles[eventName] = []
    this.handles[eventName].push(callback)
    if(vm instanceOf this.Vue) this.setEventMapUid(vm._uid, eventName)
  }
  
  $emit(){
    let args = [...arguments]
    let eventName = args[0]
    let params = args.slice(1)
    if(this.handles[eventName]){
      let len = this.handles[eventName].length
      for(let i =0; i< len; i++){
        this.handles[eventName][i](...params)
      }
    }
  }
  
  $offVmEvent(uid){
    let currentEvents = this.eventMapUid[uid] || []
    currentEvents.forEach(event => {
      this.$off(event)
    })
  }
  
  $off(eventName){
    delete this.handles[eventName]
  }
}

// vue 插件
let $EventBus = {}
$EventBus.install = (Vue, option) => {
  Vue.prototype.$eventBus = new EventBus(Vue)
  Vue.mixin({
    beforeDestroy(){
      this.$eventBus.$offVmEvent(this._uid)
    }
  })
}

export default $EventBus
```
